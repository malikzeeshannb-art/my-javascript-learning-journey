Day 5 — JavaScript Data Types Practice

Author: Zeeshan

1) STRING

// Strings store text inside quotes
let myName = "Zeeshan";
let address = "Pakistan";
console.log("My name is:", myName);
console.log("I live in:", address);

2) NUMBER

// Numbers store numeric values (whole or decimal)
let age = 23;
let salary = 10000.54;
let price = 5.50;
console.log("Age:", age);
console.log("Salary:", salary);
console.log("Price:", price);

3) BOOLEAN

// Boolean stores true or false
let isLearningJs = true;
let isTodayDay5 = true;
let isSkyBlue = true;
console.log("Learning JS:", isLearningJs);
console.log("Is it Day 5:", isTodayDay5);
console.log("Is the sky blue:", isSkyBlue);

4) UNDEFINED

// Declared variables without value are undefined
let futureGoal;
let pastActivity;
console.log("Future goal:", futureGoal);
console.log("Past activity:", pastActivity);

5) NULL

// null is an empty value assigned intentionally
let marriedStatus = null;
let owenHouse = null;
let iAmSuccessful = null;
console.log("Married status:", marriedStatus);
console.log("Own house:", owenHouse);
console.log("Am I successful:", iAmSuccessful);

6) OBJECT

// Objects store multiple values in key-value pairs
let userPhone = {
  existingFacebook: true,
  existingInstagram: true,
  existingPaidVpn: false,
  existingPubgMobile: null,
  existingXxxVideos: false
};
console.log("User Phone Data:", userPhone);
